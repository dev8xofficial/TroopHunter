# To stop all running containers, use the following command:
docker stop $(docker ps -a -q)

# If you wish to remove all stopped containers, use:
docker rm $(docker ps -a -q)

# To kill single container
docker ps
docker kill [container_id]

# docker command to remove containers
docker system prune -f

# docker command to remove images
docker image prune -a -f
docker rmi $(docker images -q)

# To know which port is using which application or service in Mac
lsof -i -n -P | grep LISTEN

# ==Start== Below instructions automate ubuntu server process

# Step 1: Don't use bridge network mode in ubuntu server
# Because it cause errors while downloading backend Dockerfile stuff.

# Step 2: To allow user password free access
# Run below commands in ubuntu server
sudo visudo
ubuntu-desktop ALL=(ALL) NOPASSWD:ALL

# Step 3: To add user in sudo group
# Run below commands in ubuntu server
sudo usermod -aG sudo ubuntu-desktop

# Step 4: To copy public key in ubuntu server
# Run below commands in ansible docker terminal
ssh-copy-id -i secrets/id_ed25519_ubuntu ubuntu-desktop@192.168.1.17

# ==End==
